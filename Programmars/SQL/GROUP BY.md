```Oracle
# 고양이와 개는 몇 마리 있을까
SELECT ANIMAL_TYPE, COUNT(*) count
FROM ANIMAL_INS
GROUP BY ANIMAL_TYPE 
ORDER BY ANIMAL_TYPE;
# 동명 동물 수 찾기
SELECT NAME,COUNT(*) COUNT
FROM ANIMAL_INS
WHERE NAME IS NOT NULL
GROUP BY NAME
HAVING COUNT(*)>=2
ORDER BY NAME;
# 입양 시각 구하기 (1)
-- 코드를 입력하세요
SELECT HOUR,COUNT(*) COUNT
FROM (SELECT TO_CHAR (DATETIME,'HH24') HOUR
     FROM ANIMAL_OUTS)
WHERE HOUR>=9 AND HOUR<20
GROUP BY HOUR
ORDER BY HOUR;

# 입양 시각 구하기 (2)
SELECT R.lv, NVL(E.cnt,0) 
    FROM (SELECT TO_CHAR(DATETIME,'HH24') as HOUR ,COUNT(*) cnt 
            FROM ANIMAL_OUTS GROUP BY TO_CHAR(DATETIME,'HH24'
                ORDER BY HOUR)E, 
        (SELECT (LEVEL-1)lv FROM dual CONNECT BY LEVEL <=24)R
WHERE R.lv = E.HOUR(+) ORDER BY R.lv
```
